    <script>
        document.addEventListener('DOMContentLoaded', (event) => {
            gsap.registerPlugin(ScrollTrigger);

            // 1. Animation Hero Load (Séquence)
            const heroTimeline = gsap.timeline();
            heroTimeline
                .from(".hero-el", {
                    opacity: 0,
                    y: 20,
                    duration: 0.6,
                    stagger: 0.2, // Délai entre les éléments
                    ease: "power2.out"
                }, 0.5) // Démarre 0.5s après le chargement

            // 2. Animation ScrollTrigger pour les Sections (Cartes Projets)
            gsap.from(".project-card", {
                opacity: 0,
                y: 50,
                stagger: 0.2,
                duration: 0.8,
                ease: "power2.out",
                scrollTrigger: {
                    trigger: "#projets",
                    start: "top 75%", // Commence l'animation quand la section est à 75% du viewport
                }
            });
            
            // 3. Animation ScrollTrigger pour les Items de Stack
            gsap.from(".stack-item", {
                opacity: 0,
                scale: 0.8,
                stagger: 0.1,
                duration: 0.5,
                ease: "back.out(1.7)",
                scrollTrigger: {
                    trigger: "#stack",
                    start: "top 80%",
                }
            });

            // 4. Animation ScrollTrigger pour la Timeline
            gsap.from(".timeline-item", {
                opacity: 0,
                y: 30,
                stagger: 0.4,
                duration: 0.8,
                ease: "power2.out",
                scrollTrigger: {
                    trigger: ".timeline",
                    start: "top 85%",
                }
            });

            // 5. Animation de Comptage des Statistiques
            const statItems = document.querySelectorAll('.stat-item p[data-target]');
            statItems.forEach(item => {
                const targetValue = parseInt(item.getAttribute('data-target'));
                const isPercentage = item.textContent.includes('%');
                const isPlus = item.textContent.includes('+');

                // Utilisation de ScrollTrigger pour démarrer le comptage
                ScrollTrigger.create({
                    trigger: item,
                    start: "top 85%",
                    onEnter: () => {
                        gsap.to(item, {
                            duration: 2,
                            // Utilisation d'un objet proxy pour animer la valeur
                            onUpdate: function() {
                                const value = Math.floor(this.targets()[0].innerHTML);
                                let display = value.toString();
                                
                                // Ajout des suffixes
                                if (isPercentage) {
                                    display += '%';
                                } else if (isPlus) {
                                    display += '+';
                                }
                                
                                item.textContent = display;
                            },
                            // Remplacer l'animation par la valeur cible
                            innerHTML: targetValue,
                            ease: "power1.out",
                            snap: { innerHTML: 1 } // S'assurer que le nombre est entier
                        });
                    },
                    once: true // Ne s'exécute qu'une seule fois
                });
            });


        });
    </script>
